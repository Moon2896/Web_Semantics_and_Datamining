requetes sparql

PREFIX webSem: <http://www.semanticweb.org/marya/ontologies/2023/2/webSem#>

1.List the instances of the class Actor

SELECT ?name
     WHERE { ?x rdf:type webSem:Actor.
             ?x  webSem:name ?name }

2. List the name of all Thriller movies. For each one, display its director.

SELECT ?title ?director
                   WHERE { ?x webSem:wasDirectedBy ?y.
                           ?y webSem:name ?director.
                           ?x webSem:title ?title.
                           ?x  webSem:genre "Thriller" }

3. List the name of all Crime Thriller movies.

SELECT ?title ?genre
                    WHERE { ?x webSem:title ?title.
                            ?x  webSem:genre "Thriller".
                            ?x webSem:genre "Crime".
                            ?x webSem:genre ?genre
                            }

4. List the name of Actors older than 51 years.

SELECT ?name ?age
                    WHERE { ?x rdf:type webSem:Actor.
                             ?x  webSem:name ?name.
                             ?x webSem:age ?age 
                             FILTER (?age>51)}

5. List of movies that are played in theater for a specific day and where and until when

SELECT ?title ?theater ?start ?end
                    WHERE { ?x rdf:type webSem:Movie.
                            ?x webSem:title ?title.
                             ?x  webSem:inTheDatesOf ?y.
                             ?z webSem:hasBroadcastTimeOf ?y.
                             ?z webSem:theaterName ?theater.
                             ?y webSem:startDate ?start.
                             ?y webSem:endDate ?end
                             }

6. A query that contains at least 2 Optional Graph Patterns

SELECT ?name ?age ?nationality
                    WHERE { ?x rdf:type webSem:Person.
                             ?x  webSem:name ?name.
                             OPTIONAL{?x webSem:age ?age.
                                        ?x webSem:nationalty ?nationality}
                             }

8. A query that contains a CONSTRUCT query form

CONSTRUCT { ?x webSem:name ?name }
                WHERE {?x webSem:name ?name }

9. A query that contains an ASK query form

ASK { ?x webSem:name "Brad Pitt" }
